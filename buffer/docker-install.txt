** Ensure there is no previous docker installaiton on the box. Stop, remove and clean up all, if needed.
   Remove any users/groups created for docker
**

As root do the following steps
1- Add the following to /etc/yum.repos.d/docker.repo
[dockerrepo]
name=dockerrepo
baseurl=https://yum.dockerproject.org/repo/main/centos/7
enabled=1
gpgcheck=1
gpgkey=https://yum.dockerproject.org/gpg
proxy=http://proxyprd.scotia-capital.com:8080

2- install docker
yum install docker-engine.x86_64

3- create artifactory folder
mkdir -p /etc/docker/certs.d/af.cds.bns:5002
chmod -R 755 /etc/docker/certs.d/af.cds.bns\:5002
4- Edit the file /etc/docker/certs.d/af.cds.bns:5002/ca.crt
vi /etc/docker/certs.d/af.cds.bns\:5002/ca.crt

5- add proxy
mkdir -p /etc/systemd/system/docker.service.d
vi /etc/systemd/system/docker.service.d/http-proxy.conf
[Service]
Environment="HTTP_PROXY=http://proxyprd.scotia-capital.com:8080/" "HTTPS_PROXY=http://gateway.bns:8000" "NO_PROXY=localhost,127.0.0.0/8,af.cds.bns"

6- restart
systemctl enable docker
systemctl daemon-reload
systemctl start docker

usermod -aG docker jenkins
